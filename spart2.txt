let actx=null,masterGain=null,comp=null;
let leftOsc=null,rightOsc=null,harmOsc=null;

function audioInit(){
  if(actx) return true;
  try{
    actx=new(window.AudioContext||window.webkitAudioContext)();
    comp=actx.createDynamicsCompressor();
    comp.threshold.value=-12;comp.ratio.value=4;
    masterGain=actx.createGain();
    masterGain.gain.setValueAtTime(S.volume*0.26,actx.currentTime);
    masterGain.connect(comp);comp.connect(actx.destination);
    return true;
  }catch(e){return false;}
}

function audioStart(){

  audioStop();
  const f=FREQUENCIES[S.freqIdx];if(!f) return;
  const base=f.hz<20?200:Math.min(f.hz,1200);
  const beat=getBeatHz();

  const lo=actx.createOscillator(),lg=actx.createGain();
  lo.frequency.value=base;lo.type='sine';lg.gain.value=0.3*S.binInt;
  lo.connect(lg);lg.connect(merge,0,0);
  const ro=actx.createOscillator(),rg=actx.createGain();
  ro.frequency.value=base+beat;ro.type='sine';rg.gain.value=0.3*S.binInt;
  ro.connect(rg);rg.connect(merge,0,1);

  ho.type='triangle';ho.frequency.value=base*1.5;hg.gain.value=0.06*S.harmDepth;

}

   Rich layered synthesis with ADSR, harmonics, tremolo, reverb-delay
═══════════════════════════ */
const toneNodes={};// {idx: {nodes:[], gain:GainNode, timer:setInterval, startTime}}

function buildToneAudio(idx){
  if(!audioInit()) return false;

  const td=TONE_DATA[idx];
  const hz=td.hz;
  const baseHz=hz<20?200:hz;
  const nodes=[];

  // Master gain for this tone with ADSR
  const toneGain=actx.createGain();
  toneGain.gain.setValueAtTime(0,actx.currentTime);
  toneGain.gain.linearRampToValueAtTime(S.toneVol*0.22,actx.currentTime+2.5); // attack

  // Simple reverb via delay feedback network
  const delay=actx.createDelay(2.0);
  delay.delayTime.value=0.12+S.toneRev*0.4;
  const delayGain=actx.createGain();
  delayGain.gain.value=S.toneRev*0.35;
  delay.connect(delayGain);
  delayGain.connect(delay);
  delayGain.connect(toneGain);

  // LFO for tremolo warmth
  const lfo=actx.createOscillator();
  const lfoGain=actx.createGain();
  lfo.frequency.value=0.08+Math.random()*0.04;
  lfoGain.gain.value=0.04;
  lfo.connect(lfoGain);
  lfo.start();
  nodes.push(lfo);

  function makeLayer(freq,type,gainVal,detune=0){
    const o=actx.createOscillator();
    const g=actx.createGain();
    o.type=type; o.frequency.value=freq; o.detune.value=detune;
    g.gain.value=gainVal;
    lfoGain.connect(g.gain); // tremolo modulation
    o.connect(g); g.connect(delay); g.connect(toneGain);
    o.start();
    nodes.push(o);
    return o;
  }

  // Layer 1: fundamental (sine — pure)
  makeLayer(baseHz,'sine',0.5);
  // Layer 2: octave up (triangle — warm)
  makeLayer(baseHz*2,'triangle',0.18*S.toneHarm,-2);
  // Layer 3: perfect 5th (sine — harmonic)
  makeLayer(baseHz*1.5,'sine',0.12*S.toneHarm,1);
  // Layer 4: octave+fifth (triangle — bright)
  makeLayer(baseHz*3,'triangle',0.06*S.toneHarm,-1);
  // Layer 5: sub-octave (sine — depth)
  if(baseHz>40) makeLayer(baseHz*0.5,'sine',0.08*S.toneHarm,0);

  // Binaural: right channel slight offset for theta entrainment
  const binOsc=actx.createOscillator();
  const binGain=actx.createGain();
  binOsc.type='sine'; binOsc.frequency.value=baseHz+7; binGain.gain.value=0.08;
  binOsc.connect(binGain); binGain.connect(toneGain);
  binOsc.start(); nodes.push(binOsc);

  toneGain.connect(masterGain);

  return{nodes,gain:toneGain,delay,delayGain,lfoGain,startTime:actx.currentTime};
}

function playTone(idx){
  if(!audioInit()) return;
  if(actx.state==='suspended') actx.resume();

  // If already playing, stop it
  if(toneNodes[idx]){
    stopTone(idx); return;
  }

  const result=buildToneAudio(idx);
  if(!result) return;

  toneNodes[idx]=result;

  // Timer
  result.timer=setInterval(()=>{
    const el=document.getElementById('tone-timer-'+idx);
    if(el){
      const elapsed=actx.currentTime-result.startTime;
      el.textContent=fmtTime(Math.round(elapsed));
    }
  },1000);

  S.activeToneIdx=idx;
  updateToneCardState(idx,true);
  toast(`♪ ${TONE_DATA[idx].hz} Hz — ${TONE_DATA[idx].name}`,'tt');
}

function stopTone(idx){
  const tn=toneNodes[idx];if(!tn) return;
  clearInterval(tn.timer);
  const now=actx.currentTime;
  // Fade out
  tn.gain.gain.setValueAtTime(tn.gain.gain.value,now);
  tn.gain.gain.linearRampToValueAtTime(0,now+1.8);
  setTimeout(()=>{
    tn.nodes.forEach(n=>{try{n.stop();n.disconnect();}catch(e){}});
    try{tn.gain.disconnect();tn.delay.disconnect();tn.delayGain.disconnect();}catch(e){}
    delete toneNodes[idx];
  },2000);
  if(S.activeToneIdx===idx) S.activeToneIdx=-1;
  updateToneCardState(idx,false);
}

function stopAllTones(){
  Object.keys(toneNodes).forEach(idx=>stopTone(parseInt(idx)));
  toast('■ All tones stopped');
}

const scopeAnimations={};

function startScopeAnim(idx){
  if(scopeAnimations[idx]) return;
  const cv=document.getElementById('tc-scope-'+idx);if(!cv) return;
  const ctx=cv.getContext('2d');
  let t=0;
  const td=TONE_DATA[idx];
  function draw(){
    if(!toneNodes[idx]){scopeAnimations[idx]=null;return;}
    cv.width=cv.parentElement?.offsetWidth-28||200;
    cv.height=40;
    const W=cv.width,H=cv.height;
    ctx.clearRect(0,0,W,H);
    ctx.fillStyle='rgba(4,11,18,0.5)';ctx.fillRect(0,0,W,H);
    // Multi-harmonic wave
    ctx.beginPath();
    for(let x=0;x<W;x++){
      const n=x/W;
      const y=H/2+(
        Math.sin(n*Math.PI*20+t*0.06)*0.5+
        Math.sin(n*Math.PI*40+t*0.04)*0.2+
        Math.sin(n*Math.PI*60+t*0.03)*0.12
      )*H*0.38;
      x===0?ctx.moveTo(x,y):ctx.lineTo(x,y);
    }
    ctx.strokeStyle=td.color+'cc';ctx.lineWidth=1.5;
    ctx.shadowBlur=6;ctx.shadowColor=td.color;ctx.stroke();ctx.shadowBlur=0;
    t++;
    scopeAnimations[idx]=requestAnimationFrame(draw);
  }
  draw();
}

function stopScopeAnim(idx){
  if(scopeAnimations[idx]){cancelAnimationFrame(scopeAnimations[idx]);scopeAnimations[idx]=null;}
  const cv=document.getElementById('tc-scope-'+idx);if(!cv) return;
  const ctx=cv.getContext('2d');

  ctx.strokeStyle=TONE_DATA[idx].color+'44';ctx.lineWidth=1;ctx.stroke();
}

function handleTonePlay(idx){
  if(toneNodes[idx]){
    stopTone(idx);
    stopScopeAnim(idx);
  }else{
    playTone(idx);
    startScopeAnim(idx);
    // Also lock session frequency to this tone
    const fIdx=FREQUENCIES.findIndex(f=>f.hz===TONE_DATA[idx].hz);
    if(fIdx>=0) selectFreq(fIdx);
  }
}
